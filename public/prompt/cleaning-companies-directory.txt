**GEOLOCATION INDEXES:**
```sql
-- PostgreSQL with PostGIS extension
CREATE EXTENSION IF NOT EXISTS postgis;

-- Add geography column for efficient distance queries
ALTER TABLE "CleaningCompany" 
ADD COLUMN location geography(POINT, 4326);

-- Update location based on lat/lng
UPDATE "CleaningCompany" 
SET location = ST_SetSRID(ST_MakePoint(longitude, latitude), 4326);

-- Create spatial index
CREATE INDEX idx_company_location ON "CleaningCompany" USING GIST(location);

-- Distance query example (find companies within 25 miles)
SELECT *, 
  ST_Distance(location, ST_SetSRID(ST_MakePoint($longitude, $latitude), 4326)) as distance
FROM "CleaningCompany"
WHERE ST_DWithin(
  location, 
  ST_SetSRID(ST_MakePoint($longitude, $latitude), 4326),
  40233.6  -- 25 miles in meters
)
ORDER BY distance;
```
