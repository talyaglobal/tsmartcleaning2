**FileUpload Component (Reusable):**
```javascript
function FileUpload({ name, accept, maxSize, maxFiles, onUpload, files, onRemove, required }) {
  const [dragActive, setDragActive] = useState(false);
  const inputRef = useRef(null);
  
  const handleDrag = (e) => {
    e.preventDefault();
    e.stopPropagation();
    if (e.type === "dragenter" || e.type === "dragover") {
      setDragActive(true);
    } else if (e.type === "dragleave") {
      setDragActive(false);
    }
  };
  
  const handleDrop = (e) => {
    e.preventDefault();
    e.stopPropagation();
    setDragActive(false);
    
    if (e.dataTransfer.files && e.dataTransfer.files[0]) {
      handleFiles(e.dataTransfer.files);
    }
  };
  
  const handleChange = (e) => {
    e.preventDefault();
    if (e.target.files && e.target.files[0]) {
      handleFiles(e.target.files);
    }
  };
  
  const handleFiles = (fileList) => {
    const filesArray = Array.from(fileList);
    
    // Validate files
    const validFiles = filesArray.filter(file => {
      if (file.size > maxSize) {
        alert(`${file.name} is too large. Max size is ${formatFileSize(maxSize)}`);
        return false;
      }
      return true;
    });
    
    if (files.length + validFiles.length > maxFiles) {
      alert(`You can only upload up to ${maxFiles} files`);
      return;
    }
    
    onUpload(name, validFiles);
  };
  
  return (
    <div
      className={`relative border-2 border-dashed rounded-lg p-8 text-center transition-colors ${
        dragActive ? 'border-blue-500 bg-blue-50' : 'border-gray-300 bg-gray-50'
      }`}
      onDragEnter={handleDrag}
      onDragLeave={handleDrag}
      onDragOver={handleDrag}
      onDrop={handleDrop}
    >
      <input
        ref={inputRef}
        type="file"
        name={name}
        accept={accept}
        multiple={maxFiles > 1}
        onChange={handleChange}
        className="hidden"
        required={required && (!files || files.length === 0)}
      />
      
      <Upload className="w-12 h-12 mx-auto text-gray-400 mb-4" />
      
      <p className="text-gray-700 mb-2">
        <strong>Drag and drop</strong> your files here, or{' '}
        <button
          type="button"
          onClick={() => inputRef.current?.click()}
          className="text-blue-600 hover:text-blue-700 underline"
        >
          browse
        </button>
      </p>
      
      <p className="text-sm text-gray-500">
        {accept} (max {formatFileSize(maxSize)} per file)
      </p>
    </div>
  );
}
```

```javascript
function useFormValidation(step, formData) {
  const [errors, setErrors] = useState({});
  
  const validateStep = () => {
    const schema = getSchemaForStep(step);
    const result = schema.safeParse(formData);
    
    if (!result.success) {
      const newErrors = {};
      result.error.errors.forEach(error => {
        newErrors[error.path[0]] = error.message;
      });
      setErrors(newErrors);
      return false;
    }
    
    setErrors({});
    return true;
  };
  
  return { errors, validateStep, setErrors };
}

function getSchemaForStep(step) {
  const schemas = {
    1: step1Schema,
    2: step2Schema,
    3: step3Schema,
    4: step4Schema,
    5: step5Schema,
    6: step6Schema,
    7: step7Schema,
    8: step8Schema,
  };
  return schemas[step] || z.object({});
}
```

function useAutoSave(formData, applicationId) {
  const [lastSaved, setLastSaved] = useState(null);
  const [isSaving, setIsSaving] = useState(false);
  
  useEffect(() => {
    // Auto-save every 30 seconds
    const interval = setInterval(() => {
      saveDraft();
    }, 30000);
    
    return () => clearInterval(interval);
  }, [formData]);
  
  const saveDraft = async () => {
    setIsSaving(true);
    try {
      // Save to localStorage
      localStorage.setItem('jobApplication', JSON.stringify({
        ...formData,
        lastModified: new Date().toISOString(),
        applicationId: applicationId || generateId()
      }));
      
      // Optional: Save to backend
      if (applicationId) {
        await fetch('/api/applications/draft', {
          method: 'POST',
          headers: { 'Content-Type': 'application/json' },
          body: JSON.stringify({ applicationId, formData })
        });
      }
      
      setLastSaved(new Date());
    } catch (error) {
      console.error('Failed to save draft:', error);
    } finally {
      setIsSaving(false);
    }
  };
  
  const loadDraft = () => {
    const saved = localStorage.getItem('jobApplication');
    if (saved) {
      return JSON.parse(saved);
    }
    return null;
  };
  
  const deleteDraft = () => {
    localStorage.removeItem('jobApplication');
  };
  
  return { saveDraft, loadDraft, deleteDraft, lastSaved, isSaving };
}

```javascript
async function handleSubmit(formData) {
  try {
    // Show loading state
    setIsSubmitting(true);
    
    // Create FormData for file uploads
    const submitData = new FormData();
    
    // Append all form fields
    Object.keys(formData).forEach(key => {
      if (key.includes('Files')) {
        // Handle file uploads
        formData[key]?.forEach(file => {
          submitData.append(key, file);
        });
      } else if (Array.isArray(formData[key])) {
        // Handle arrays
        submitData.append(key, JSON.stringify(formData[key]));
      } else if (typeof formData[key] === 'object' && formData[key] !== null) {
        // Handle objects
        submitData.append(key, JSON.stringify(formData[key]));
      } else {
        // Handle primitives
        submitData.append(key, formData[key]);
      }
    });
    
    // Submit to backend
    const response = await fetch('/api/applications/submit', {
      method: 'POST',
      body: submitData
    });
    
    if (!response.ok) {
      throw new Error('Submission failed');
    }
    
    const result = await response.json();
    
    // Clear draft
    deleteDraft();
    
    // Redirect to success page
    router.push(`/apply/success?id=${result.applicationId}`);
    
  } catch (error) {
    console.error('Submission error:', error);
    setSubmitError('Failed to submit application. Please try again.');
  } finally {
    setIsSubmitting(false);
  }
}
```

```javascript
// pages/api/applications/submit.js
import { IncomingForm } from 'formidable';
import { promises as fs } from 'fs';
import path from 'path';

export const config = {
  api: {
    bodyParser: false,
  },
};

export default async function handler(req, res) {
  if (req.method !== 'POST') {
    return res.status(405).json({ message: 'Method not allowed' });
  }
  
  try {
    // Parse form data
    const form = new IncomingForm({
      uploadDir: path.join(process.cwd(), 'uploads'),
      keepExtensions: true,
      maxFileSize: 50 * 1024 * 1024, // 50MB
    });
    
    const { fields, files } = await new Promise((resolve, reject) => {
      form.parse(req, (err, fields, files) => {
        if (err) reject(err);
        resolve({ fields, files });
      });
    });
    
    // Generate application ID
    const applicationId = generateApplicationId();
    
    // Process files (upload to S3, Cloudinary, etc.)
    const processedFiles = await processUploadedFiles(files, applicationId);
    
    // Save to database
    await saveApplication({
      applicationId,
      ...fields,
      files: processedFiles,
      status: 'pending',
      submittedAt: new Date(),
    });
    
    // Send confirmation email
    await sendConfirmationEmail(fields.email, applicationId);
    
    // Notify admins
    await notifyAdmins(applicationId);
    
    res.status(200).json({
      success: true,
      applicationId,
      message: 'Application submitted successfully'
    });
    
  } catch (error) {
    console.error('Submission error:', error);
    res.status(500).json({
      success: false,
      message: 'Failed to process application'
    });
  }
}
```
